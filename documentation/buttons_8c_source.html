40 int8_t buttons_event(uint8_t pin) {
41  static uint8_t buttonState = 0xff;
42  if (_BV(pin) & BUTTON_MASK) {
43  uint8_t s = buttons_get(pin);
44  if (s != ((buttonState >> pin) & 1)) {
45  buttonState &= ~(1 << pin);
46  buttonState |= s << pin;
47  if (s == 1)
48  return 1;
49  }

51  }
